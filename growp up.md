export type ContentFormat = 'reel_b_roll' | 'static_dark_mode' | 'carousel_slide' | 'blog_post';

interface MotivationalGenerationContext {
  category: string; // 'monk_mode', 'discipline', 'biohacking', 'wealth_automation', 'stoicism'
  themeDescription: string;
  contentType: ContentFormat; 
  aesthetic: MonkModeAesthetic;
  language?: 'english' | 'hindi' | 'marathi';
  recentQuotes: string; 
  targetAudience?: 'entrepreneurs' | 'students' | 'men_20s' | 'general';
}

interface GeneratedMotivationalContent {
  hookText: string; // The "Stop Scrolling" text (Overlay)
  captionBody: string; // The deep value/lesson
  cta: string; // "Comment MONK for the protocol"
  visualPrompt: string; // Highly specific AI prompt
  hexPalette: string; // Specific colors to enforce
  audioMood: string; // 'Phonk', 'Ambient', 'Dark Synth'
  hashtags: string;
  // Blog specific fields (optional)
  blogTitle?: string;
  blogOutline?: string;
  affiliateAngle?: string;
}

export const MotivationalPromptRefinerService = {
  
  /**
   * Main generation function routing to specific handlers based on content type
   */
  async generateUniqueContent(context: MotivationalGenerationContext): Promise<GeneratedMotivationalContent> {
    // If it's a blog request, route to the blog generator
    if (context.contentType === 'blog_post') {
      return this.generateBlogContent(context);
    }
    
    // Otherwise handle Social Media (Video/Image)
    return this.generateSocialContent(context);
  },

  /**
   * Generates High-Retention Social Media Content (Reels/Static)
   */
  async generateSocialContent(context: MotivationalGenerationContext): Promise<GeneratedMotivationalContent> {
    try {
      const modelName = getTextModelName();
      const uniqueSeed = `${Date.now()}-${Math.random().toString(36).substring(7)}`;

      // 1. Define the 2026 Aesthetic Logic
      const aestheticRules = {
        monk_midnight: {
          colors:,
          visual_keywords: 'luxury, deep navy background, champagne gold text, cinematic lighting, 8k, sharp focus',
          mood: 'Authority, Wealth, Stoic'
        },
        gen_x_soft_club: {
          colors: ['#000000', '#1A1A2E', '#4A4E69'],
          visual_keywords: 'aggressive motion blur, light trails, digital noir, CRT monitor glow, night city, lonely, focus',
          mood: 'Tech, Deep Work, Isolation'
        },
        plum_noir: {
          colors:,
          visual_keywords: 'velvet texture, deep red shadows, marble statues, renaissance lighting, dramatic contrast',
          mood: 'Philosophy, Pain, Sacrifice'
        },
        raw_authentic: {
          colors: ['#1A1A1A', '#333333', '#F2F2F2'],
          visual_keywords: 'handheld camera movement, 35mm film grain, slightly out of focus, low light, messy desk, gym sweat, reality',
          mood: 'Grind, Real, Unfiltered'
        },
        olive_spruce: {
          colors:,
          visual_keywords: 'forest fog, rain on window, nature, muted greens, organic texture, solitude',
          mood: 'Growth, Peace, Grounding'
        }
      };

      const currentStyle = aestheticRules[context.aesthetic];

      // 2. Construct the "Negative Framing" Prompt
      // In 2026, we avoid generic motivation. We use "Negative/Warning" hooks.
      const prompt = `
      You are a specialized Content Architect for the "Monk Mode" niche in 2026.
      SEED: ${uniqueSeed}
      
      TARGET AUDIENCE: ${context.targetAudience |

| 'men seeking discipline'}
      CATEGORY: ${context.category}
      AESTHETIC: ${context.aesthetic} (Vibe: ${currentStyle.mood})
      
      YOUR GOAL: Generate content that STOPS the scroll using "Negative Framing" or "Curiosity Gaps".
      Avoid generic "You can do it" quotes. Use "Hard Truths", "Contrarian Views", or "Wake Up Calls".

      INPUT CONTEXT: ${context.themeDescription}

      GENERATE JSON with this structure:
      {
        "hookText": "Short, punchy overlay text (Max 10 words). Use Negative Framing (e.g., 'Stop doing this', 'Why you are failing').",
        "captionBody": "A 2-3 sentence profound caption expanding on the hook. Stoic, direct, no fluff.",
        "cta": "A specific Call to Action related to the topic (e.g., 'Comment FOCUS for the protocol').",
        "visualPrompt": "A highly detailed AI image/video generation prompt. MUST include: ${currentStyle.visual_keywords}. For video, specify camera movement.",
        "suggestedAudio": "Genre suggestions (e.g., Phonk, Lo-Fi, Spoken Word)",
        "hashtags": "Mix of 3 broad, 5 niche, 2 community tags."
      }
      `;

      const response = await genAI.models.generateContent({
        model: modelName,
        contents: prompt,
      });

      // Parse and processing logic (similar to your original, but stricter)
      const text = response.candidates?.?.content?.parts?.?.text;
      if (!text) throw new Error('No content generated');
      
      const jsonMatch = text.match(/\{*\}/);
      if (!jsonMatch) throw new Error('Invalid JSON');
      
      const parsed = JSON.parse(jsonMatch);

      // 3. Post-Process Visual Prompt with Hardcoded Colors
      // We inject the hex codes directly to ensure the AI video/image generator adheres to the brand palette.
      let enhancedVisualPrompt = parsed.visualPrompt;
      const colorString = currentStyle.colors.join(' and ');
      
      if (context.contentType === 'reel_b_roll') {
        enhancedVisualPrompt = `Cinematic vertical video (9:16), ${currentStyle.visual_keywords}, color palette dominated by ${colorString}. ${enhancedVisualPrompt}. No text on screen. Raw, authentic, high contrast.`;
      } else {
        enhancedVisualPrompt = `Minimalist dark aesthetic image (4:5), ${currentStyle.visual_keywords}, main colors ${colorString}. ${enhancedVisualPrompt}. Text placement area: Center.`;
      }

      return {
        hookText: parsed.hookText,
        captionBody: parsed.captionBody,
        cta: parsed.cta,
        visualPrompt: enhancedVisualPrompt,
        hexPalette: currentStyle.colors,
        audioMood: parsed.suggestedAudio |

| 'Slowed + Reverb',
        hashtags: parsed.hashtags
      };

    } catch (error) {
      console.error("Content Gen Error", error);
      // Fallback to a safe "Monk Mode" default
      return {
        hookText: "Discipline > Motivation.",
        captionBody: "Stop waiting for the right feeling. Do the work.",
        cta: "Comment 'MONK' to start.",
        visualPrompt: "Dark luxury office, night time, rain on window, deep navy and gold lighting, cinematic 8k",
        hexPalette:,
        audioMood: 'Dark Phonk',
        hashtags: '#monkmode #discipline #2026'
      };
    }
  },

  /**
   * Generates Viral Blog Content (SEO + Affiliate Optimized)
   */
  async generateBlogContent(context: MotivationalGenerationContext): Promise<GeneratedMotivationalContent> {
    const modelName = getTextModelName();
    
    // Blog logic focuses on "The Digital Stoic" voice: Direct, Authoritative, Second-Person ("You").
    // It creates an outline designed for "Skim-reading" (Bionic reading principles).
    const blogPrompt = `
    Generate a high-converting Blog Post structure for the "Monk Mode 2026" niche.
    TOPIC: ${context.themeDescription}
    CATEGORY: ${context.category}
    
    REQUIREMENTS:
    1. Title: Must be a "How-To" or "Listicle" with a curiosity gap (e.g., "The 4 AM Protocol: Why I Wake Up When You Sleep").
    2. Tone: Direct, Stoic, Authoritative. No fluff.
    3. Structure: 4-5 main headers (H2).
    4. Affiliate Integration: Suggest where to place affiliate links (e.g., Notion Templates, Oura Ring, Supplements).
    
    Output JSON:
    {
      "blogTitle": "Catchy Title",
      "hookText": "The meta-description/intro hook",
      "blogOutline":,
      "affiliateAngle": "Specific advice on what products fit this article",
      "visualPrompt": "Prompt for the blog header image (Dark Mode style)",
      "hashtags": "SEO Keywords"
    }
    `;

    const response = await genAI.models.generateContent({
      model: modelName,
      contents: blogPrompt,
    });

    const text = response.candidates?.?.content?.parts?.?.text;
    const jsonMatch = text?.match(/\{*\}/);
    if (!jsonMatch) throw new Error('Blog generation failed');
    
    const parsed = JSON.parse(jsonMatch);

    return {
      hookText: parsed.hookText,
      captionBody: "Full blog content generation would happen in a separate stream.", // Placeholder for full body
      cta: "Link in bio to read full protocol.",
      visualPrompt: `Wide aspect ratio (16:9) blog header, ${context.aesthetic} style, ${parsed.blogTitle} theme, dark mode UI, minimalist.`,
      hexPalette:, // Default to Monk Midnight for blogs
      audioMood: 'Lo-Fi Focus',
      hashtags: parsed.hashtags,
      blogTitle: parsed.blogTitle,
      blogOutline: parsed.blogOutline,
      affiliateAngle: parsed.affiliateAngle
    };
  }
};
